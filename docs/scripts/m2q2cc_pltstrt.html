void main()
{
    object oPC = GetPCSpeaker();
    int iJaxState = GetLocalInt(oPC, "NW_JOURNAL_ENTRYm2q2_Jax");

    SetLocalInt(GetPCSpeaker(),"M2Q2CC_L" + GetTag(OBJECT_SELF),1);

    if (iJaxState > 0 && iJaxState < 36)
    {
        AddJournalQuestEntry("m2q2_Jax", 36, oPC);
    }
}
