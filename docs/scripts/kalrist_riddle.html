int CheckIsInInventory (object oChest, object oGem)
{
    object oObj = GetFirstItemInInventory(oChest);
    while (GetIsObjectValid(oObj))
    {
        if (oGem == oObj)
        {return TRUE;}
        oObj = GetNextItemInInventory(oChest);

    return FALSE;
}
