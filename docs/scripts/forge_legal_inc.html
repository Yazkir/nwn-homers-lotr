// Fixed-width fingerprint tuple for one permanent property; padding makes
// the NWScript lexicographic sort agree with the generator's numeric sort.
// Normalization must mirror gen-forge-legal.py's normalize_prop().
string ForgeLegalPad5(int n)
{
    if (n < 0) n = 65535; // engine signals "none" as -1; GFF stores 65535
    string s = IntToString(n);
    while (GetStringLength(s) < 5)
        s = "0" + s;
    return s;
}

// TRUE when fixed-width tuple sA sorts before sB. NWScript has no string
// ordering operators, so compare the four zero-padded numeric fields —
// equivalent to the generator's lexicographic sort on these strings.
int ForgeLegalTupleLess(string sA, string sB)
{
    int i;
    for (i = 0; i < 4; i++)
    {
        int nA = StringToInt(GetSubString(sA, i * 6, 5));
        int nB = StringToInt(GetSubString(sB, i * 6, 5));
        if (nA != nB)
            return nA < nB;
    }
    return FALSE;
}

// Runtime fingerprint of oItem: resref|t:s:c:p,... ascending, permanent
// properties only (matches ForgeCountProps). Insertion sort via string scan.
string ForgeLegalFingerprint(object oItem)
{
    string sBody = "";
    itemproperty ip = GetFirstItemProperty(oItem);
    while (GetIsItemPropertyValid(ip))
    {
        // Cosmetic props (visual effects, Light) are excluded, matching
        // forge_inc's ForgeIsCosmeticProp and the generator.
        int nPropType = GetItemPropertyType(ip);
        if (GetItemPropertyDurationType(ip) == DURATION_TYPE_PERMANENT
            && nPropType != ITEM_PROPERTY_VISUALEFFECT
            && nPropType != ITEM_PROPERTY_LIGHT)
        {
            int nP1 = GetItemPropertyParam1(ip);
            int nPV;
            if (nP1 < 0 || nP1 == 255)
                nPV = 65535; // no param1 slot on this property
            else
                nPV = GetItemPropertyParam1Value(ip);
            string sTuple = ForgeLegalPad5(GetItemPropertyType(ip))
                + ":" + ForgeLegalPad5(GetItemPropertySubType(ip))
                + ":" + ForgeLegalPad5(GetItemPropertyCostTableValue(ip))
                + ":" + ForgeLegalPad5(nPV);
            // insert sTuple into sBody keeping ascending order
            int nLen = GetStringLength(sBody);
            int nTuple = GetStringLength(sTuple);
            int nPos = 0;
            int bPlaced = FALSE;
            string sNew = "";
            while (nPos < nLen)
            {
                string sAt = GetSubString(sBody, nPos, nTuple);
                if (!bPlaced && ForgeLegalTupleLess(sTuple, sAt))
                {
                    sNew += sTuple + ",";
                    bPlaced = TRUE;
                }
                sNew += sAt + ",";
                nPos += nTuple + 1;
            }
            if (!bPlaced)
                sNew += sTuple + ",";
            sBody = sNew;
        }
        ip = GetNextItemProperty(oItem);
    }
    // strip trailing comma
    int nBody = GetStringLength(sBody);
    if (nBody > 0)
        sBody = GetStringLeft(sBody, nBody - 1);
    return GetResRef(oItem) + "|" + sBody;
}
