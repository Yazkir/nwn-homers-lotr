{
   int nMatch = GetListenPatternNumber();
   object oSpeaker = GetLastSpeaker();
   if(GetLocalObject(OBJECT_SELF, "DM_SUMMONER") == oSpeaker)
   {
       if(nMatch == 8686)
       {

       }
   }
}
